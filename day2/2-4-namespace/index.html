    2-4. Namespace мӢӨмҠө¶

    мӢӨмҠө лӘ©н‘ң¶

    • Namespaceмқҳ м—ӯн• кіј н•„мҡ”м„ұмқ„ мқҙн•ҙн•ңлӢӨ.
    • лӘ…л №м–ҙмҷҖ YAMLлЎң NamespaceлҘј мғқм„ұн•ҳкі  кҙҖлҰ¬н•ңлӢӨ.
    • кё°ліё м»Ён…ҚмҠӨнҠё NamespaceлҘј ліҖкІҪн•ҳлҠ” л°©лІ•мқ„ мқөнһҢлӢӨ.
    • Namespace мӮӯм ң мӢң лӮҙл¶Җ лҰ¬мҶҢмҠӨк°Җ н•Ёк»ҳ мӮӯм ңлҗЁмқ„ нҷ•мқён•ңлӢӨ.

    NamespaceлһҖ?¶

    Kubernetes нҒҙлҹ¬мҠӨн„° м•Ҳм—җм„ң лҰ¬мҶҢмҠӨлҘј л…јлҰ¬м ҒмңјлЎң кІ©лҰ¬н•ҳлҠ” лӢЁмң„мһ…лӢҲлӢӨ.

    нҒҙлҹ¬мҠӨн„°
в”ңв”Җв”Җ default          вҶҗ м•„л¬ҙ м„Өм • м—Ҷмқҙ л°°нҸ¬н•ҳл©ҙ м—¬кё° л“Өм–ҙк°җ
в”ңв”Җв”Җ kube-system      вҶҗ K8s мӢңмҠӨн…ң м»ҙнҸ¬л„ҢнҠё
в”ңв”Җв”Җ dev              вҶҗ к°ңл°ң нҷҳкІҪ
в””в”Җв”Җ prod             вҶҗ мҡҙмҳҒ нҷҳкІҪ
    • к°ҷмқҖ Namespace м•Ҳм—җм„ңлҠ” Service мқҙлҰ„мңјлЎң м„ңлЎң нҶөмӢ  к°ҖлҠҘ
    • лӢӨлҘё Namespaceмқҳ лҰ¬мҶҢмҠӨлҠ” кё°ліём ҒмңјлЎң кІ©лҰ¬лҗЁ
    • NamespaceлҘј мӮӯм ңн•ҳл©ҙ к·ё м•Ҳмқҳ лӘЁл“  лҰ¬мҶҢмҠӨк°Җ н•Ёк»ҳ мӮӯм ңлҗЁ

    1) кё°ліё Namespace нҷ•мқё¶

    kubectl get namespaces    # м•Ҫм–ҙ: kubectl get ns

    м¶ңл Ҙ мҳҲмӢң:

    NAME              STATUS   AGE
default           Active   5d
kube-system       Active   5d
kube-public       Active   5d
kube-node-lease   Active   5d
    Namespace мҡ©лҸ„
    default м•„л¬ҙ Namespace м§Җм • м—Ҷмқҙ мғқм„ұн•ҳл©ҙ м—¬кё° л°°нҸ¬лҗЁ
    kube-system CoreDNS, metrics-server л“ұ мӢңмҠӨн…ң м»ҙнҸ¬л„ҢнҠё
    kube-public мқёмҰқ м—Ҷмқҙ мқҪмқ„ мҲҳ мһҲлҠ” кіөк°ң лҰ¬мҶҢмҠӨ
    kube-node-lease л…ёл“ң heartbeat лҰ¬мҶҢмҠӨ

    нҠ№м • Namespaceмқҳ лҰ¬мҶҢмҠӨлҘј ліҙл Өл©ҙ -n н”Ңлһҳк·ёлҘј мӮ¬мҡ©н•©лӢҲлӢӨ.

    kubectl get pods -n kube-system

    2) лӘ…л №м–ҙлЎң Namespace мғқм„ұ¶

    kubectl create namespace dev
kubectl get ns              # dev мғқм„ұ нҷ•мқё

    Namespace м•Ҳм—җ PodлҘј л°°нҸ¬н•ҳл Өл©ҙ -n н”Ңлһҳк·ёлҘј мӮ¬мҡ©н•©лӢҲлӢӨ.

    kubectl run nginx-dev --image=nginx:1.25 -n dev
kubectl get pods -n dev

    лӘЁл“  Namespaceмқҳ PodлҘј н•ңкәјлІҲм—җ ліҙл Өл©ҙ --all-namespaces(м•Ҫм–ҙ -A)лҘј мӮ¬мҡ©н•©лӢҲлӢӨ.

    kubectl get pods -A

    3) YAMLлЎң Namespace мғқм„ұ¶

    namespace-prod.yaml нҢҢмқјмқ„ л§Ңл“ӯлӢҲлӢӨ.

    apiVersion: v1
kind: Namespace
metadata:
  name: prod
  labels:
    env: production

    kubectl apply -f namespace-prod.yaml
kubectl get ns prod

    PodлҸ„ YAML м•Ҳм—җ namespaceлҘј лӘ…мӢңн•ҙм„ң л°°нҸ¬н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    apiVersion: v1
kind: Pod
metadata:
  name: nginx-prod
  namespace: prod
spec:
  containers:
    - name: nginx
      image: nginx:1.25

    kubectl apply -f namespace-prod.yaml   # мқҙлҜё мһҲм–ҙлҸ„ мҳӨлҘҳ м—ҶмқҢ
kubectl get pods -n prod

    4) кё°ліё Namespace м „нҷҳ¶

    л§ӨлІҲ -n devлҘј л¶ҷмқҙлҠ” лҢҖмӢ  кё°ліё NamespaceлҘј л°”кҝҖ мҲҳ мһҲмҠөлӢҲлӢӨ.

    kubectl config set-context --current --namespace=dev

    kubectl config view --minify | grep namespace    # нҳ„мһ¬ кё°ліё Namespace нҷ•мқё

    kubectl config view --minify | Select-String "namespace"    # нҳ„мһ¬ кё°ліё Namespace нҷ•мқё

    м „нҷҳ нӣ„м—җлҠ” -n м—ҶмқҙлҸ„ dev Namespaceм—җ лӘ…л №мқҙ м Ғмҡ©лҗ©лӢҲлӢӨ.

    kubectl get pods    # dev Namespaceмқҳ Podк°Җ м¶ңл ҘлҗЁ

    мӢӨмҠөмқҙ лҒқлӮҳл©ҙ л°ҳл“ңмӢң defaultлЎң ліөмӣҗн•©лӢҲлӢӨ.

    kubectl config set-context --current --namespace=default
kubectl get pods    # default Namespaceмқҳ Podк°Җ м¶ңл ҘлҗЁ

    Namespace м „нҷҳ нӣ„ кјӯ ліөмӣҗн•ҳм„ёмҡ”

    кё°ліё NamespaceлҘј л°”кҫј мұ„лЎң л‘җл©ҙ мқҙнӣ„ мӢӨмҠөм—җм„ң лҰ¬мҶҢмҠӨк°Җ м—үлҡұн•ң Namespaceм—җ л°°нҸ¬лҗ  мҲҳ мһҲмҠөлӢҲлӢӨ. мӢӨмҠө нӣ„ л°ҳл“ңмӢң defaultлЎң лҗҳлҸҢлҰ¬м„ёмҡ”.

    5) Namespace мӮӯм ң вҖ” лӮҙл¶Җ лҰ¬мҶҢмҠӨ н•Ёк»ҳ мӮӯм ң¶

    NamespaceлҘј мӮӯм ңн•ҳл©ҙ к·ё м•Ҳмқҳ лӘЁл“  лҰ¬мҶҢмҠӨ(Pod, Service, Deployment л“ұ)к°Җ н•Ёк»ҳ мӮӯм ңлҗ©лӢҲлӢӨ.

    лЁјм Җ dev Namespace м•Ҳмқҳ лҰ¬мҶҢмҠӨлҘј нҷ•мқён•©лӢҲлӢӨ.

    kubectl get all -n dev

    NamespaceлҘј мӮӯм ңн•©лӢҲлӢӨ.

    kubectl delete namespace dev

    мӮӯм ң нӣ„ лҰ¬мҶҢмҠӨк°Җ мӮ¬лқјмЎҢлҠ”м§Җ нҷ•мқён•©лӢҲлӢӨ.

    kubectl get pods -n dev    # Error from server (NotFound) м¶ңл Ҙ

    Namespace мӮӯм ңлҠ” лҗҳлҸҢлҰҙ мҲҳ м—ҶмҠөлӢҲлӢӨ

    Namespace мӮӯм ң мӢң лӮҙл¶Җмқҳ лӘЁл“  лҰ¬мҶҢмҠӨк°Җ мҰүмӢң мӮӯм ңлҗ©лӢҲлӢӨ. мҡҙмҳҒ нҷҳкІҪм—җм„ң мӢӨмҲҳлЎң мӮӯм ңн•ҳм§Җ м•ҠлҸ„лЎқ мЈјмқҳн•ҳм„ёмҡ”.

    м •лҰ¬¶

    kubectl delete namespace prod
kubectl get ns    # dev, prod лӘЁл‘җ мӮӯм ң нҷ•мқё

    Namespace лӘ…л №м–ҙ м •лҰ¬¶

    лӘ…л №м–ҙ м„ӨлӘ…
    kubectl get ns Namespace лӘ©лЎқ мЎ°нҡҢ
    kubectl create ns <мқҙлҰ„> Namespace мғқм„ұ
    kubectl get pods -n <мқҙлҰ„> нҠ№м • Namespaceмқҳ Pod мЎ°нҡҢ
    kubectl get pods -A м „мІҙ Namespaceмқҳ Pod мЎ°нҡҢ
    kubectl config set-context --current --namespace=<мқҙлҰ„> кё°ліё Namespace ліҖкІҪ
    kubectl delete namespace <мқҙлҰ„> Namespace мӮӯм ң (лӮҙл¶Җ лҰ¬мҶҢмҠӨ нҸ¬н•Ё)

    нҠёлҹ¬лё”мҠҲнҢ…¶

    мҰқмғҒ нҷ•мқё мӮ¬н•ӯ
    Podк°Җ ліҙмқҙм§Җ м•ҠмқҢ нҳ„мһ¬ кё°ліё Namespace нҷ•мқё (kubectl config view --minify)
    Service мқҙлҰ„мңјлЎң нҶөмӢ  м•Ҳ лҗЁ к°ҷмқҖ Namespaceмқём§Җ нҷ•мқё
    Namespace мӮӯм ң нӣ„ Terminating мғҒнғң м§ҖмҶҚ finalizerк°Җ кұёлҰ° лҰ¬мҶҢмҠӨ мЎҙмһ¬ вҖ” kubectl get all -n <мқҙлҰ„> мңјлЎң нҷ•мқё
