    RBAC 03 вҖ” ClusterRole & ClusterRoleBinding¶

    мӢӨмҠө лӘ©н‘ң¶

    • Role vs ClusterRole, RoleBinding vs ClusterRoleBinding м°ЁмқҙлҘј мқҙн•ҙн•ңлӢӨ.
    • нҒҙлҹ¬мҠӨн„° лІ”мң„ лҰ¬мҶҢмҠӨ(nodes, pv л“ұ)м—җ м ‘к·јн•ҳлҠ” ClusterRoleмқ„ л§Ңл“ лӢӨ.
    • 4к°Җм§Җ мЎ°н•©мқҳ м°ЁмқҙлҘј м§Ғм ‘ нҷ•мқён•ңлӢӨ.

    к°ңл…җ¶

    Role vs ClusterRole¶

    н•ӯлӘ© Role ClusterRole
    лІ”мң„ нҠ№м • Namespace нҒҙлҹ¬мҠӨн„° м „мІҙ
    м Ғмҡ© лҰ¬мҶҢмҠӨ Namespace лҰ¬мҶҢмҠӨ (Pod, Service л“ұ) нҒҙлҹ¬мҠӨн„° лҰ¬мҶҢмҠӨ (Node, PV л“ұ) + лӘЁл“  NS лҰ¬мҶҢмҠӨ
    мғқм„ұ мң„м№ҳ Namespace м•Ҳ нҒҙлҹ¬мҠӨн„° л ҲлІЁ (NS м—ҶмқҢ)

    нҒҙлҹ¬мҠӨн„° лІ”мң„ лҰ¬мҶҢмҠӨлҠ” RoleлЎң м ‘к·ј к¶Ңн•ңмқ„ мӨ„ мҲҳ м—ҶмҠөлӢҲлӢӨ.

    # нҒҙлҹ¬мҠӨн„° лІ”мң„ лҰ¬мҶҢмҠӨ лӘ©лЎқ нҷ•мқё
kubectl api-resources --namespaced=false
# nodes, persistentvolumes, clusterroles, namespaces л“ұ...

    4к°Җм§Җ мЎ°н•©¶

    ClusterRole  +  ClusterRoleBinding  вҶ’  нҒҙлҹ¬мҠӨн„° м „мІҙ лІ”мң„ к¶Ңн•ң
ClusterRole  +  RoleBinding         вҶ’  нҠ№м • NamespaceлЎң лІ”мң„ м ңн•ң
Role         +  RoleBinding         вҶ’  нҠ№м • Namespace лІ”мң„ к¶Ңн•ң (кё°ліё)
Role         +  ClusterRoleBinding  вҶ’  мӮ¬мҡ© л¶Ҳк°Җ (мҳӨлҘҳ)

    мӮ¬м „ мӨҖл№„¶

    kubectl create namespace ns-a
kubectl create namespace ns-b
kubectl create serviceaccount cluster-reader -n ns-a

    Step 1. ClusterRole мғқм„ұ¶

    л…ёл“ңмҷҖ PV(нҒҙлҹ¬мҠӨн„° лІ”мң„ лҰ¬мҶҢмҠӨ)лҘј мЎ°нҡҢн•ҳлҠ” ClusterRoleмқ„ л§Ңл“ӯлӢҲлӢӨ.

    clusterrole-reader.yaml:

    apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: cluster-resource-reader
rules:
  - apiGroups: [""]
    resources: ["nodes", "persistentvolumes", "namespaces"]
    verbs: ["get", "list", "watch"]
  - apiGroups: [""]
    resources: ["pods"]
    verbs: ["get", "list", "watch"]

    kubectl apply -f clusterrole-reader.yaml
kubectl get clusterrole cluster-resource-reader
kubectl describe clusterrole cluster-resource-reader

    Step 2. ClusterRoleBindingмңјлЎң м—°кІ° вҶ’ нҒҙлҹ¬мҠӨн„° м „мІҙ к¶Ңн•ң¶

    clusterrolebinding-reader.yaml:

    apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: cluster-resource-reader-binding
subjects:
  - kind: ServiceAccount
    name: cluster-reader
    namespace: ns-a
roleRef:
  kind: ClusterRole
  name: cluster-resource-reader
  apiGroup: rbac.authorization.k8s.io

    kubectl apply -f clusterrolebinding-reader.yaml

    Step 3. нҒҙлҹ¬мҠӨн„° лІ”мң„ к¶Ңн•ң нҷ•мқё¶

    # л…ёл“ң мЎ°нҡҢ (нҒҙлҹ¬мҠӨн„° лҰ¬мҶҢмҠӨ)
kubectl auth can-i list nodes \
  --as=system:serviceaccount:ns-a:cluster-reader
# yes

# PV мЎ°нҡҢ
kubectl auth can-i list persistentvolumes \
  --as=system:serviceaccount:ns-a:cluster-reader
# yes

# ns-bмқҳ Pod мЎ°нҡҢ (лӢӨлҘё NS)
kubectl auth can-i list pods \
  --as=system:serviceaccount:ns-a:cluster-reader \
  -n ns-b
# yes вҶҗ нҒҙлҹ¬мҠӨн„° м „мІҙ лІ”мң„мқҙлҜҖлЎң лӢӨлҘё NSлҸ„ к°ҖлҠҘ

    Step 4. ClusterRole + RoleBinding вҶ’ Namespace лІ”мң„лЎң м ңн•ң¶

    к°ҷмқҖ ClusterRoleмқҙм§Җл§Ң мқҙлІҲм—” RoleBindingмңјлЎң м—°кІ°н•ҳл©ҙ ns-aм—җм„ңл§Ң мң нҡЁн•©лӢҲлӢӨ.

    # кё°мЎҙ ClusterRoleBinding м ңкұ°
kubectl delete clusterrolebinding cluster-resource-reader-binding

    rolebinding-cluster-role.yaml:

    apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: cluster-role-ns-binding
  namespace: ns-a          # ns-aм—җл§Ң м Ғмҡ©
subjects:
  - kind: ServiceAccount
    name: cluster-reader
    namespace: ns-a
roleRef:
  kind: ClusterRole        # ClusterRole м°ёмЎ° к°ҖлҠҘ
  name: cluster-resource-reader
  apiGroup: rbac.authorization.k8s.io

    kubectl apply -f rolebinding-cluster-role.yaml

    к¶Ңн•ң мһ¬нҷ•мқё:

    # ns-aмқҳ Pod мЎ°нҡҢ
kubectl auth can-i list pods \
  --as=system:serviceaccount:ns-a:cluster-reader \
  -n ns-a
# yes вҶҗ ns-aлҠ” к°ҖлҠҘ

# ns-bмқҳ Pod мЎ°нҡҢ
kubectl auth can-i list pods \
  --as=system:serviceaccount:ns-a:cluster-reader \
  -n ns-b
# no вҶҗ RoleBindingмқҙлқјм„ң ns-aлЎң м ңн•ңлҗЁ

# л…ёл“ң мЎ°нҡҢ (нҒҙлҹ¬мҠӨн„° лҰ¬мҶҢмҠӨ вҖ” NS л¬ҙкҙҖ)
kubectl auth can-i list nodes \
  --as=system:serviceaccount:ns-a:cluster-reader
# no вҶҗ RoleBindingмқҖ нҒҙлҹ¬мҠӨн„° лҰ¬мҶҢмҠӨм—җ нҡЁл Ҙ м—ҶмқҢ

    н•өмӢ¬ м°Ёмқҙ

    ClusterRoleмқ„ ClusterRoleBindingмңјлЎң м—°кІ° вҶ’ нҒҙлҹ¬мҠӨн„° м „мІҙ м Ғмҡ© ClusterRoleмқ„ RoleBindingмңјлЎң м—°кІ° вҶ’ н•ҙлӢ№ NamespaceлЎң м ңн•ң

    Step 5. лӮҙмһҘ ClusterRole нҷңмҡ©¶

    мҝ лІ„л„ӨнӢ°мҠӨм—җлҠ” кё°ліё м ңкіө ClusterRoleмқҙ мһҲмҠөлӢҲлӢӨ. м§Ғм ‘ л§Ңл“Өм§Җ м•Ҡкі  мһ¬мӮ¬мҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    kubectl get clusterrole | grep -E "^view|^edit|^admin|^cluster-admin"

    kubectl get clusterrole | Select-String "^view|^edit|^admin|^cluster-admin"
    лӮҙмһҘ ClusterRole м„ӨлӘ…
    view лҢҖл¶Җл¶„ лҰ¬мҶҢмҠӨ мқҪкё° м „мҡ©
    edit лҢҖл¶Җл¶„ лҰ¬мҶҢмҠӨ мқҪкё°/м“°кё° (Role мҲҳм • л¶Ҳк°Җ)
    admin Namespace лӮҙ кұ°мқҳ лӘЁл“  к¶Ңн•ң
    cluster-admin нҒҙлҹ¬мҠӨн„° м „мІҙ лӘЁл“  к¶Ңн•ң 
    # лӮҙмһҘ view ClusterRoleмқ„ нҠ№м • NSм—җ м Ғмҡ©
kubectl create rolebinding ns-a-view \
  --clusterrole=view \
  --serviceaccount=ns-a:cluster-reader \
  -n ns-a

kubectl auth can-i list pods \
  --as=system:serviceaccount:ns-a:cluster-reader \
  -n ns-a
# yes
